async function saveProduct() { const btn = event.target; btn.disabled = true; try { let title = document.getElementById('prodTitle').value.trim(); let content = document.getElementById('prodContent').value.trim(); if (!title || !content) return alert('标题和内容不能为空'); let res = await fetch('/api/products', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': 'ok' }, body: JSON.stringify({ title, content }) }); let text = await res.text(); if (!res.ok) throw new Error(text); alert('保存成功'); document.getElementById('prodTitle').value = ''; document.getElementById('prodContent').value = ''; loadProducts(); } catch (e) { alert('保存失败：' + e.message); } finally { btn.disabled = false; } } async function saveNews() { const btn = event.target; btn.disabled = true; try { let title = document.getElementById('newsTitle').value.trim(); let date = document.getElementById('newsDate').value; let content = document.getElementById('newsContent').value.trim(); if (!title || !date || !content) return alert('所有字段不能为空'); let res = await fetch('/api/news', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': 'ok' }, body: JSON.stringify({ title, date, content }) }); let text = await res.text(); if (!res.ok) throw new Error(text); alert('保存成功'); document.getElementById('newsTitle').value = ''; document.getElementById('newsDate').value = ''; document.getElementById('newsContent').value = ''; loadNews(); } catch (e) { alert('保存失败：' + e.message); } finally { btn.disabled = false; } }
